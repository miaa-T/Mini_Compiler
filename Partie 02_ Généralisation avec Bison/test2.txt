fn int factorial(int n) {    //mahdia
    int result ;;
    result = 1;;
    int i;;
    for (i from 1 to n) {
        result = result * i;;
    };;
    giveback result}

fn float power(float base, int exponent) {    ///  djihene
    float result = 1.0;;
    while (exponent > 0) repeat {
        result = result * base;;
        exponent = exponent - 1;;
    };;
    giveback result
}
